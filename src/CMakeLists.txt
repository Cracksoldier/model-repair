add_library(modelrepair SHARED
    Mesh.cpp
    RepairPipeline.cpp
    RepairReport.cpp
    repairs/MergeDuplicateVertices.cpp
    repairs/RemoveDegenerateTriangles.cpp
    repairs/FixNonManifold.cpp
    repairs/FixNormals.cpp
    repairs/FillHoles.cpp
    repairs/RemoveSelfIntersections.cpp
    io/StlIO.cpp
    io/ObjIO.cpp
    io/ThreeMFIO.cpp
    io/GlbIO.cpp
    io/MeshIO.cpp
)

target_include_directories(modelrepair
    PUBLIC
        $<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>
)

target_include_directories(modelrepair
    PRIVATE
        # lib3mf only sets PRIVATE includes on its own target, so we add the
        # C++ bindings header path explicitly.
        ${lib3mf_SOURCE_DIR}/Autogenerated/Bindings/Cpp
)

target_link_libraries(modelrepair
    PUBLIC
        CGAL::CGAL
        CGAL::CGAL_Core
        Eigen3::Eigen
    PRIVATE
        lib3mf
        tinygltf
        modelrepair_warnings
)

set_target_properties(modelrepair PROPERTIES
    VERSION   ${PROJECT_VERSION}
    SOVERSION ${PROJECT_VERSION_MAJOR}
)
